\section*{}
\begin{enumerate}
  \item Ofen auf 160°C vorwärmen
  \item 2 Geschirre mit Butter fetten
  \item Rosmarin + 2 KL Vollrahm pro Geschirr
  \item Je ein Ei ins Geschirr schlagen
  \item Mit Salz + Pfeffer (+ Paprika) würzen
  \item 14-16min Backen
  \item Wenn Eiweiss fast hart ist und Eigelb noch weich rausnehmen
\end{enumerate}
